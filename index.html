📋 Database Setup Guide
Step-by-step:
  1. Go to supabase.com → Create a new project
  2. Open SQL Editor in the left sidebar
  3. Copy the SQL below → Paste into the editor → Click Run
  4. Go to Project Settings → API
  5. Copy your Project URL and anon public key
  6. Paste both above and click Test & Connect
-- ============================================================
-- SANDUGO MULTIPURPOSE COOPERATIVE — Full Database Schema
-- Run this entire script in Supabase SQL Editor
-- ============================================================

-- 1. USERS TABLE
create table if not exists users (
  username        text primary key,
  password        text not null,
  role            text not null default 'member',
  fname           text,
  lname           text,
  email           text,
  joined          text,
  status          text default 'active',
  share_capital   numeric default 0,
  is_old_member   boolean default false
);

-- 2. COLLECTIONS TABLE
create table if not exists collections (
  id        bigint primary key,
  username  text references users(username) on delete cascade,
  amount    numeric not null,
  year      text,
  date      text,
  note      text,
  type      text default 'collection'
);

-- 3. LOANS TABLE
create table if not exists loans (
  id               bigint primary key,
  username         text references users(username) on delete cascade,
  amount           numeric not null,
  repaid           numeric default 0,
  date             text,
  due_date         text,
  interest         numeric default 18,
  monthly_payment  numeric,
  total_payable    numeric,
  note             text,
  status           text default 'active',
  payments         jsonb default '[]',
  processing_fee   numeric default 0,
  sc_allotment     numeric default 0,
  net_release      numeric,
  from_application bigint,
  is_imported      boolean default false
);

-- 4. LOAN APPLICATIONS TABLE
create table if not exists loan_applications (
  id               bigint primary key,
  username         text references users(username) on delete cascade,
  submitted_at     text,
  name             text,
  gender           text,
  age              int,
  address          text,
  mobile           text,
  income_source    text,
  monthly_income   numeric,
  monthly_expenses numeric,
  loan_amount      numeric,
  purpose          text,
  notes            text,
  status           text default 'pending',
  reviewed_at      text,
  loan_id          bigint
);

-- ============================================================
-- ROW LEVEL SECURITY (required by Supabase)
-- ============================================================
alter table users             enable row level security;
alter table collections       enable row level security;
alter table loans             enable row level security;
alter table loan_applications enable row level security;

-- Allow all operations (adjust for production as needed)
create policy "Allow all" on users             for all using (true) with check (true);
create policy "Allow all" on collections       for all using (true) with check (true);
create policy "Allow all" on loans             for all using (true) with check (true);
create policy "Allow all" on loan_applications for all using (true) with check (true);

-- 5. DATABASE CREDENTIALS TABLE
create table if not exists db_credentials (
  id         serial primary key,
  db_user    text not null unique,
  db_pass    text not null,
  label      text default 'Main Connection',
  created_at text
);
alter table db_credentials enable row level security;
create policy "Allow all" on db_credentials for all using (true) with check (true);

-- ============================================================
-- INDEXES for performance
-- ============================================================
create index if not exists idx_collections_username on collections(username);
create index if not exists idx_loans_username        on loans(username);
create index if not exists idx_loans_status          on loans(status);
create index if not exists idx_apps_username         on loan_applications(username);
create index if not exists idx_apps_status           on loan_applications(status);

📋 loan_applications
id, username, submitted_at, name, gender, age, address, mobile, income_source, monthly_income, monthly_expenses, loan_amount, purpose, notes, status, reviewed_at, loan_id
🔑 db_credentials
id (serial), db_user (unique), db_pass, label, created_at
